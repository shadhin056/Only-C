#include<stdio.h>
#include<math.h>

main ()

{
    int n = 1;

    while(n <= 100)
    {
        printf("%d\n", n);
        n++;
        if(n > 10)
        {
            break;
        }
    }
    printf("\n\n\n");
    return 0;

}
