    int a,b,c,d,pa,pb,pc,pd;

        for(b=2; b<a; b++)      //a,b,c> 1
        {
            pb=b*b*b;
            for(c=b; c<a; c++)
            {
                pc=c*c*c;
                for(d=c; d<a; d++)
                {
                    pd=d*d*d;
                    if(pa==pb+pc+pd)
                    {
                        printf("Cube = %d, Triple = (%d,%d,%d)\n",a,b,c,d);
                        break;
                    }
                }
            }
        }
    }

    return 0;
