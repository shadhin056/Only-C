int main()
{
    freopen("uvainput.txt", "r", stdin);
    //freopen("uvaoutput.txt", "w", stdout);


    long int a,l,n,c,k=0;

    while(scanf("%ld %ld",&a,&l)==2)
    {
        k++;
        if(a<0 && l<0)
            break;

        n=a;
        c=1;

        while(n!=1)
        {
            if(n%2!=0)
                n=3*n+1;
            else
                n=n/2;
            if(n>l)
                break;
            c++;
        }
        printf("Case %ld: A = %ld, limit = %ld, number of terms = %ld\n"
               ,k,a,l,c);
    }

    return 0;

}
