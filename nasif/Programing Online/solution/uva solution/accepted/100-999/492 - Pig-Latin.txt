        if(isvowel(ch))

            while(true)
            {
                scanf("%c",&ch);
                if(isalpha(ch))

                {
                    printf("ay%c",ch);

        }

            tch=ch;

}

int isvowel(char ch)
{
    if(ch=='a'||ch=='e'||ch=='i'||ch=='o'||ch=='u'||
       ch=='A'||ch=='E'||ch=='I'||ch=='O'||ch=='U')
        return 1;
    return 0;

}
