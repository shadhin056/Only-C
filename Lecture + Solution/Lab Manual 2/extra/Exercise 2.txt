#include<stdio.h>

main()
{
    float height, length, width, volume;

    printf("enter height of the box= ");
    scanf("%f", &height);
    printf("enter length of the box= ");
    scanf("%f", &length);
    printf("enter width of the box= ");
    scanf("%f", &width);

    volume = height*length*width;
    printf("\n\n\nvolume=%\n\n\n,volume");
}
